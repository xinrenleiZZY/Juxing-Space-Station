## 二、API配置

### 2.1 创建API工具

```javascript
// utils/api.js
import axios from 'axios'

const API_BASE_URL = process.env.NUXT_PUBLIC_API_URL || 'http://localhost:8000'

export const apiClient = axios.create({
  baseURL: API_BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
apiClient.interceptors.request.use(
  (config) => {
    // 添加认证token
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
apiClient.interceptors.response.use(
  (response) => {
    return response.data
  },
  (error) => {
    if (error.response?.status === 401) {
      // 处理未授权错误
      localStorage.removeItem('token')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

// API方法
export const evaluationApi = {
  // 上传文档
  uploadDocument: async (formData) => {
    return apiClient.post('/api/documents/upload', formData, {
      headers: {
        'Content-Type': 'multipart/form-data'
      }
    })
  },
  
  // 启动评估
  startEvaluation: async (data) => {
    return apiClient.post('/api/evaluations/start', data)
  },
  
  // 获取评估结果
  getEvaluation: async (evaluationId) => {
    return apiClient.get(`/api/evaluations/${evaluationId}`)
  },
  
  // 获取评估历史
  getEvaluationHistory: async (params) => {
    return apiClient.get('/api/evaluations', { params })
  }
}

export const chatApi = {
  // 智能问答
  chat: async (data) => {
    return apiClient.post('/api/chat', data)
  },
  
  // 获取对话历史
  getChatHistory: async (conversationId) => {
    return apiClient.get(`/api/chat/history/${conversationId}`)
  }
}
```

### 2.2 创建Composables

```javascript
// composables/useApi.js
import { evaluationApi, chatApi } from '@/utils/api'

export const useApi = () => {
  const isLoading = ref(false)
  const error = ref(null)
  
  const execute = async (apiFunc, ...args) => {
    isLoading.value = true
    error.value = null
    
    try {
      const result = await apiFunc(...args)
      return result
    } catch (err) {
      error.value = err.response?.data?.detail || err.message
      throw err
    } finally {
      isLoading.value = false
    }
  }
  
  return {
    isLoading,
    error,
    execute
  }
}
```

  if (!inputMessage.value.trim() || isTyping.value) return

### 4.1 评估Composable

```javascript
// composables/useEvaluation.js
import { ref } from 'vue'
import { evaluationApi } from '@/utils/api'
import { useApi } from './useApi'

export const useEvaluation = () => {
  const { isLoading: isEvaluating, error, execute } = useApi()
  const evaluationResult = ref(null)
  const documentId = ref(null)

  const startEvaluation = async (formData) => {
    try {
      // 1. 上传文档
      const uploadFormData = new FormData()
      uploadFormData.append('file', formData.file)
      uploadFormData.append('company_name', formData.companyName)
      uploadFormData.append('data_name', formData.dataName)
      uploadFormData.append('data_type', formData.dataType)
      uploadFormData.append('data_scale', formData.dataScale)
      uploadFormData.append('data_source', formData.dataSource)
      uploadFormData.append('industry', formData.industry)

      const uploadResult = await execute(
        evaluationApi.uploadDocument,
        uploadFormData
      )

      documentId.value = uploadResult.document_id

      // 2. 启动评估
      const evaluationResult = await execute(
        evaluationApi.startEvaluation,
        {
          document_id: uploadResult.document_id,
          additional_info: {
            data_description: formData.dataDescription
          }
        }
      )

      return evaluationResult
    } catch (err) {
      console.error('评估失败:', err)
      throw err
    }
  }

  const getEvaluation = async (evaluationId) => {
    return await execute(evaluationApi.getEvaluation, evaluationId)
  }

  return {
    isEvaluating,
    error,
    evaluationResult,
    documentId,
    startEvaluation,
    getEvaluation
  }
}
```

### 4.2 聊天Composable

```javascript
// composables/useChat.js
import { ref } from 'vue'
import { chatApi } from '@/utils/api'
import { useApi } from './useApi'

export const useChat = () => {
  const { isLoading: isTyping, error, execute } = useApi()
  const messages = ref([])
  const conversationId = ref(null)

  const sendMessage = async (message) => {
    // 添加用户消息
    messages.value.push({
      role: 'user',
      content: message,
      timestamp: new Date().toISOString()
    })

    try {
      // 发送消息到API
      const result = await execute(chatApi.chat, {
        question: message,
        conversation_id: conversationId.value
      })

      // 更新会话ID
      conversationId.value = result.conversation_id

      // 添加助手消息
      messages.value.push({
        role: 'assistant',
        content: result.answer,
        sources: result.sources,
        timestamp: new Date().toISOString()
      })

      return result
    } catch (err) {
      console.error('发送消息失败:', err)
      // 添加错误消息
      messages.value.push({
        role: 'assistant',
        content: '抱歉，我遇到了一些问题。请稍后再试。',
        isError: true,
        timestamp: new Date().toISOString()
      })
      throw err
    }
  }

  const clearMessages = () => {
    messages.value = []
    conversationId.value = null
  }

  return {
    messages,
    isTyping,
    error,
    conversationId,
    sendMessage,
    clearMessages
  }
}
```
